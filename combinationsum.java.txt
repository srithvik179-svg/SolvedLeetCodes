class Combinationsum {
    public List<List<Integer>> combinationSum(int[] arr, int target) {
 List<Integer>current=new ArrayList<>();
 List<List<Integer>> result = new ArrayList<>();
 int i=0;
 solve(i,arr,target,result,current);
 return result;
 }
 private static void solve(int i,int[] arr,int target,List<List<Integer>> result,List<Integer>current){
    if(target==0){
        result.add(new ArrayList<Integer>(current));
        return;
    }
    if(target<0||i==arr.length) return;
    current.add(arr[i]);
    solve(i,arr,target-arr[i],result,current);
    current.remove(current.size()-1);
    solve(i+1,arr,target,result,current);

    }
}
